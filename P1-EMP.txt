create table employee(empno number primary key,empname varchar2(20),designation varchar2(30),gender 
varchar2(6),age number,dojdate,salary number);

desc employee; 

insert into employee values(&empno,'&empname','&designation','&gender',&age,'&doj',&salary); 

select * from employee;

COMPARISON

select * from employee where salary>30000;

select * from employee where age between 25 and 30; 

LOGICAL

select * from employee where salary<50000 and salary>30000;

select * from employee where designation='manager' or designation='admin';

SORTING:

select * from employee order by empno;

select * from employee order by empno desc; 

SET OPERATION:

select * from employee where salary>30000 union select * from employee where age between 25 
and 30; 

select * from employee where salary>30000 union all select * from employee where age between 
25 and 30;

 select * from employee where salary>30000 intersect  select * from employee where age between 
25 and 30; 

 select * from employee where salary>30000 minus select * from employee where age between 25 
and 30;
